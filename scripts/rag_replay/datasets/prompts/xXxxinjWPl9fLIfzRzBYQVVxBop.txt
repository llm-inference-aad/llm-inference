def pad_num_y(k_s):
    """Calculate padding for depthwise convolution."""
    pad_per_side = int((k_s - 1) * 0.5)
    return pad_per_side

class DW(nn.Module):
    """Depthwise Convolution Block with HardSwish Activation."""
    def __init__(self, cin, dw_s):
        super().__init__()
        self.dw = nn.Conv2d(cin, cin, dw_s, 1, pad_num_y(dw_s), groups=cin)
        self.act = nn.Hardswish()

    def forward(self, x):
        """Forward pass through the Depthwise Block."""
        x = self.dw(x)
        x = self.act(x)
        return x

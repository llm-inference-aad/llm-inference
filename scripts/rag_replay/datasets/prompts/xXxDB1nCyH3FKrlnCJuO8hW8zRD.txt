import torch
import torch.nn as nn

class SE_LN(nn.Module):
    def __init__(self, cin):
        super().__init__()
        self.gavg = nn.AdaptiveAvgPool2d((1,1))  
        self.ln = nn.LayerNorm(cin)
        self.act = nn.Sigmoid()
        
    def forward(self, x):
        y = x
        x = self.gavg(x)
        x = x.view(-1, x.size(1))
        x = self.ln(x)
        x = self.act(x)
        x = x.view(-1, x.size(1), 1, 1)
        return x*y

class Net(nn.Module):
    def __init__(self):
        super().__init__()
        self.stem = nn.Conv2d(3, 64, kernel_size=3, padding=1, bias=False)
        self.act1 = nn.SiLU()
        
        # Block 1
        self.conv1 = nn.Conv2d(64, 64, kernel_size=3, padding=1, groups=64, bias=False)
        self.bn1 = nn.BatchNorm2d(64)
        self.se1 = SE_LN(64)
        self.act2 = nn.SiLU()
        
        # Block 2 (Dilated)
        self.conv2 = nn.Conv2d(64, 128, kernel_size=3, padding=2, dilation=2, groups=64, bias=False)
        self.bn2 = nn.BatchNorm2d(128)
        self.se2 = SE_LN(128)
        self.act3 = nn.SiLU()
        
        # Block 3 (Deeper, Dilated)
        self.conv3 = nn.Conv2d(128, 256, kernel_size=3, padding=4, dilation=4, groups=128, bias=False)
        self.bn3 = nn.BatchNorm2d(256)
        self.se3 = SE_LN(256)
        self.act4 = nn.SiLU()
        
        # Final Layers
        self.gap = nn.AdaptiveAvgPool2d((1,1))
        self.fc = nn.Linear(256, 10)
    
    def forward(self, x):
        x = self.act1(self.stem(x))
        
        x = self.act2(self.bn1(self.conv1(x)))
        x = self.se1(x)
        
        x = self.act3(self.bn2(self.conv2(x)))
        x = self.se2(x)
        
        x = self.act4(self.bn3(self.conv3(x)))
        x = self.se3(x)
        
        x = self.gap(x).view(x.size(0), -1)
        x = self.fc(x)
        return x

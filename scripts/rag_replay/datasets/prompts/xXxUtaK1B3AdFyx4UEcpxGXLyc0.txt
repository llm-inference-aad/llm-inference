import torch
import torch.nn as nn
import torch.optim as optim
from torch.optim.lr_scheduler import CosineAnnealingLR
from torchvision import transforms

class ME(nn.Module):
    def __init__(self, cin, cout):
        super().__init__()
        self.maxpool = nn.MaxPool2d(2, ceil_mode=True)
        self.pw = nn.Conv2d(cin, cout, 1, 1, bias=False)
        self.bn = nn.BatchNorm2d(cout)

    def forward(self, x):
        x = self.maxpool(x)
        x = self.pw(x)
        x = self.bn(x)
        return x

# Assuming a larger network structure that incorporates ME, e.g., a simple CNN for demonstration
class CIFARNet(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size=3, padding=1)
        self.me1 = ME(64, 64)
        self.conv2 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
        self.me2 = ME(128, 128)
        self.fc = nn.Linear(128*4*4, 10) # Adjusted based on expected output size after pooling
        
    def forward(self, x):
        x = torch.relu(self.conv1(x))
        x = self.me1(x)

def train(model, device, loader, optimizer, scheduler, epoch):
    model.train()
    for batch_idx, (data, target) in enumerate(loader):

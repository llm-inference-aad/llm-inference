import torch
import torch.nn as nn

class SE_LN(nn.Module):
    """Enhanced SE_LN module combining global averaging, depthwise separable convolution, and layer normalization."""
    def __init__(self, cin):
        super().__init__()
        self.gavg = nn.AdaptiveAvgPool2d((1,1))  
        self.ln = nn.LayerNorm(cin)
        self.dwsc = nn.Sequential(
            nn.Conv2d(cin, cin, kernel_size=3, groups=cin, padding=1, bias=False),  
            nn.Conv2d(cin, cin, kernel_size=1, bias=False)  
        )
        self.act = nn.Sigmoid()
        self.bn = nn.BatchNorm2d(cin)      
        
    def forward(self, x):
        """Apply global avg pool, layer norm, dwsc, and sigmoid for channel-wise scaling."""
        x_avg = self.gavg(x)
        x_avg = x_avg.view(-1, x_avg.size(1))
        x_avg = self.ln(x_avg)
        x_avg = x_avg.view(-1, x_avg.size(1), 1, 1)
        x_avg = self.act(x_avg)
        
        x_dwsc = self.dwsc(x)
        x_dwsc = self.bn(x_dwsc)  
        x_dwsc = nn.ReLU()(x_dwsc)  
        
        return x_dwsc * x_avg 

class ResNetBlock(nn.Module):
    """Basic ResNet block with an optional shortcut and enhanced SE_LN."""
    def __init__(self, cin, cout, stride=1):
        super().__init__()
        self.conv1 = nn.Conv2d(cin, cout, kernel_size=3, padding=1, stride=stride)
        self.bn1 = nn.BatchNorm2d(cout)
        self.act1 = nn.ReLU()
        self.conv2 = nn.Conv2d(cout, cout, kernel_size=3, padding=1)
        self.bn2 = nn.BatchNorm2d(cout)
        self.act2 = nn.ReLU()
        self.se_ln = SE_LN(cout)
        if cin != cout or stride != 1:
            self.shortcut = nn.Sequential(
                nn.Conv2d(cin, cout, kernel_size=1, stride=stride),
                nn.BatchNorm2d(cout)
            )
        else:
            self.shortcut = None

    def forward(self, x):
        residual = x
        out = self.conv1(x)
        out = self.bn1(out)
        out = self.act1(out)
        out = self.conv2(out)
        out = self.bn2(out)
        out = self.se_ln(out)  # Enhanced SE_LN application
        if self.shortcut is not None:
            residual = self.shortcut(x)
        out += residual
        out = self.act2(out)
        return out

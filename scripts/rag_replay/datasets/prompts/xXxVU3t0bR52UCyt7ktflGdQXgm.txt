def calculate_sum(numbers):
    total = 0
    for num in numbers:
        if isinstance(num, (int, float)):
            total += num
    return total

numbers_list = [1, 2.5, 'a', 3]
result = calculate_sum(numbers_list)
print(result)  # Output: 6.5

    """Squeeze-and-Excitation Layer with Layer Normalization."""
    def __init__(self, num_features):
        super(SE_LN, self).__init__()
        self.avg_pool = nn.AdaptiveAvgPool2d(1)
        self.fc = nn.Linear(num_features, num_features, bias=True)

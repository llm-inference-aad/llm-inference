import numpy as np

def neural_network(X, y):
    # Initialize parameters
    input_size = X.shape[1]
    hidden_size = 32
    output_size = 1
    
    W1 = np.random.randn(input_size, hidden_size)
    b1 = np.random.randn(hidden_size)
    W2 = np.random.randn(hidden_size, output_size)
    b2 = np.random.randn(output_size)
    
    # Forward propagation
    Z1 = np.dot(X, W1) + b1
    A1 = np.tanh(Z1)
    Z2 = np.dot(A1, W2) + b2
    A2 = Z2  # Output layer (regression task)
    
    # Compute loss
    loss = np.mean((A2 - y)**2)
    
    return loss

# Example usage
X = np.array([[0.5, 0.3], [0.7, 0.8]])
y = np.array([[0.6], [0.9]])
print(neural_network(X, y))

def calculate_score(text):
    words = text.split()
    word_count = len(words)
    score = 0
    for word in words:
        if word.lower() == 'good':
            score += 1
        elif word.lower() == 'bad':
            score -= 1
    return (score / word_count) * 100

text = "This is a good sentence. Another bad example."
print(calculate_score(text))

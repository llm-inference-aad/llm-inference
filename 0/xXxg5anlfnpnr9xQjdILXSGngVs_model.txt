import numpy as np

def compute_loss(y_true, y_pred):
    return np.mean((y_true - y_pred) ** 2)

x = np.array([1, 2, 3])
y = np.array([4, 5, 6])

y_pred = x * 2 + 1
loss = compute_loss(y, y_pred)
print(f"Loss: {loss}")
